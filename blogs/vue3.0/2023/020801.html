vue3跨组件通讯（类似vue2的EventBus）

LCh2023-02-08 18:03:28vue3.0vue

vue3移除了 $on，$off 和 $once 等实例方法，vue3中使用EventBus创建全局事件侦听器的方法已经失效，我们可以使用官方推荐的库替代

# 1.安装这个库mittopen in new window

$ npm install --save mitt

# 2.封装

// src/utils/bus.ts
import mitt from 'mitt';

interface BusParams {
  $on?: any;
  $off?: any;
  $emit?: any;
}
const bus: BusParams = {};
const emitter = mitt();
bus.$on = emitter.on;
bus.$off = emitter.off;
bus.$emit = emitter.emit;
export default bus;

# 3.用法

<script lang="ts" setup>
const handleTopOperate=()=>{
  console.log('1111')
}
bus.$on('busEventHandleTopOperate',(fromTemplateData)=>{
  console.log(fromTemplateData) // {name:'luochao'}
  handleTopOperate()
});
onUnmounted(() => {
  bus.$off('busEventHandleTopOperate');
});
<script>

<script lang="ts" setup>
bus.$emit('busEventHandleTopOperate',{name:'luochao'});
<script>
